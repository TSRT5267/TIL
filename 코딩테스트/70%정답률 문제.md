## 직사각형 별찍기
```
#include <iostream>

using namespace std;

int main(void) {
    int a; // 가로
    int b; // 세로
    cin >> a >> b;
    
    for(int i=0;i<b;i++)
    {
        for(int j=0;j<a;j++) /
        {
            cout<<'*';
        }
        cout<<endl;
    }

    return 0;
}
```
