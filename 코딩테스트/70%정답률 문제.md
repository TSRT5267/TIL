## 최대공약수와 최소공배수  
```
#include <string>
#include <vector>

using namespace std;

vector<int> solution(int n, int m) {
    vector<int> answer;
    
    int a =0;
    int b =0;
    int c =0;
    
    if(n>m)
    {
        a = n;
        b = m;
    }
    else
    {
        a = m;
        b = n;
    }
    
    
     while(b != 0)
    {
        c = a % b;
        a = b;
        b = c;
    }

    return answer;
}

## 이진 변환 반복하기  
```
#include <string>
#include <vector>

using namespace std;

vector<int> solution(string s) {
    vector<int> answer(2, 0);
    int zero = 0, round = 0;
    
    while(s != "1")
    {
        string tmp = "";
        int size = 0, num;
        round++;
        
        for(int i = 0; i < s.size(); i++)
        {
            if(s[i] == '0')
                zero++;
            else
                tmp += "1";
        }
        
        num = tmp.size();
        s = "";
        while(num > 0)
        {
            s += to_string(num % 2);
            num /= 2;
        }
    }
    
    answer[0] = round;
    answer[1] = zero;
    return answer;
}
```

## 같은 숫자는 싫어 
```
#include <vector>
#include <iostream>

using namespace std;

vector<int> solution(vector<int> arr) 
{
    vector<int> answer;
    vector<int> stack;
    stack.push_back(arr[0]);
    int lastNum = arr[0];
    
    for(int i=1;i<arr.size();i++)
    {
       if(lastNum != arr[i])  
       {
          stack.push_back(arr[i]); 
          lastNum=arr[i];
       }        
    }
    
    answer = stack;
    return answer;
}
```

## 3진법 뒤집기
```
#include <string>
#include <vector>

using namespace std;

int solution(int n) {
    int answer = 0;   
    vector<int> three;
    
    while(n){
        three.push_back(n%3);
        n/=3;
    }
    
    int temp=1;
    
    for(int i=three.size()-1;i>=0;i--){
        answer+=three[i]*temp;
        temp*=3;
    }
    
    return answer;
}
```

    for (int i = 0; i <= t.length() - pLen; i++) {

## 최댓값과 최솟값
```
#include <vector>

#include <algorithm>

using namespace std;

string solution(string s) {
    string answer = "", tmp = "";
    vector<int> v;
    
    for(int i = 0; i < s.size(); ++i) {
        if(s[i] == ' ') {
            v.push_back(stoi(tmp));
            tmp = "";
        }
        else tmp += s[i];
    }
    v.push_back(stoi(tmp));
    
    answer += to_string(*min_element(v.begin(), v.end()));
    answer += ' ';
    answer += to_string(*max_element(v.begin(), v.end()));
    return answer;
}

## 최소직사각형
```
#include <string>
#include <vector>
#include <algorithm>

using namespace std;

int solution(vector<vector<int>> sizes) {
    int answer = 0;
    int maxX =0;
    int maxY =0;
    
    for(int i=0;i<sizes.size();i++)
    {
        sort(sizes[i].begin(),sizes[i].end());
        
        if(sizes[i][0]> maxX) maxX = sizes[i][0];
        if(sizes[i][1]> maxY) maxY = sizes[i][1];
    }

    answer = maxX*maxY;
    return answer;
}
```

using namespace std;

int solution(string s)
{
    string stack;

    for (char c : s)
    {
        if (!stack.empty() && stack.back() == c)
        {
            stack.pop_back();
        }
        else
        {
            stack.push_back(c);
        }
    }

    return stack.empty() ? 1 : 0;

## 문자열 내 마음대로 정렬하기
```
#include <algorithm>
#include <string>
#include <vector>
using namespace std;

int N;
bool cmp(string a, string b){
    if(a[N] == b[N]){
        return a < b;
    }
    else{
        return a[N] < b[N];
    }
}

vector<string> solution(vector<string> strings, int n) {
    N=n;
    sort(strings.begin(), strings.end(), cmp);
    return strings;
}
```

int solution(int a, int b, int n) {

## 점프와 순간 이동
```
#include <iostream>

int solution(int n)
{
    int ans = 0;    
    
    while(n > 0) // 거리를 차감방식으로 계산
    {
        if(n % 2 == 0) n /= 2; // 순간이동으로 절반씩 이동
        else

        }

## [1차] 비밀지도
```
#include <string>
#include <vector>
#include <iostream>
#include <algorithm>

using namespace std;

vector<string> solution(int n, vector<int> arr1, vector<int> arr2) {
    vector<string> answer;
    
    for(int i = 0; i < n; i++) //지도의 크기n만큼 반복
    {
        string tmp = "";
        arr1[i] = arr1[i] | arr2[i]; // 두 배열을 or 연선을 통해 하나로 합침
        
        while(tmp.size() != n) // 지도의 크기인 N 만큼 tmp를 생성
        {
            // 10진수를 2진수를 변환하는 방법을 이용
            // 0 -> 공백  
            // 1 -> # 
            if(arr1[i] % 2 == 0)
                tmp.push_back(' ');
            else
                tmp.push_back('#');
            arr1[i] /= 2;
        }
        
        reverse(tmp.begin(), tmp.end()); // 역순으로 생성됨으로 뒤집어줌
        answer.push_back(tmp);
    }
    return answer;
}
